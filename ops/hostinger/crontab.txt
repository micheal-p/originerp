# Nightly backup — install with: crontab ops/hostinger/crontab.txt (on the VPS)
# Dumps Postgres + storage files, gzips, keeps 14 days locally.
# Swap the last line for an rclone push once you've picked an offsite target
# (Backblaze B2 / another provider's object storage) — never keep backups
# only on the same VPS that would be lost in an outage.

0 3 * * * PGPASSWORD=$(grep POSTGRES_PASSWORD /opt/collarone/.env | cut -d= -f2) \
  pg_dump -h 127.0.0.1 -U postgres postgres | gzip > /opt/collarone/backups/db-$(date +\%F).sql.gz \
  && find /opt/collarone/backups -name 'db-*.sql.gz' -mtime +14 -delete

30 3 * * * tar czf /opt/collarone/backups/storage-$(date +\%F).tar.gz -C /var/lib/docker/volumes/collarone_storage-data/_data . \
  && find /opt/collarone/backups -name 'storage-*.tar.gz' -mtime +14 -delete

# once offsite target is picked, add e.g.:
# 0 4 * * * rclone sync /opt/collarone/backups remote:collarone-backups
